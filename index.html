-- Grounded Planner sync table
-- Stores every planner localStorage bucket as one JSON row.
-- Work and academic schedules live in gnd-events.
-- updated_at is the cross-device source of truth.

CREATE TABLE IF NOT EXISTS public.app_data (
  data_key TEXT PRIMARY KEY,
  data_value JSONB NOT NULL DEFAULT '{}'::JSONB,
  created_at TIMESTAMPTZ NOT NULL DEFAULT CLOCK_TIMESTAMP(),
  updated_at TIMESTAMPTZ NOT NULL DEFAULT CLOCK_TIMESTAMP(),
  revision BIGINT NOT NULL DEFAULT 1
);

-- Safe migration for older versions of this setup.
ALTER TABLE public.app_data
  ADD COLUMN IF NOT EXISTS data_value JSONB NOT NULL DEFAULT '{}'::JSONB;

ALTER TABLE public.app_data
  ADD COLUMN IF NOT EXISTS created_at TIMESTAMPTZ NOT NULL DEFAULT CLOCK_TIMESTAMP();

ALTER TABLE public.app_data
  ADD COLUMN IF NOT EXISTS updated_at TIMESTAMPTZ NOT NULL DEFAULT CLOCK_TIMESTAMP();

ALTER TABLE public.app_data
  ADD COLUMN IF NOT EXISTS revision BIGINT NOT NULL DEFAULT 1;

DELETE FROM public.app_data
WHERE data_key IS NULL OR BTRIM(data_key) = '';

UPDATE public.app_data
SET
  data_key = BTRIM(data_key),
  data_value = COALESCE(data_value, '{}'::JSONB),
  created_at = COALESCE(created_at, CLOCK_TIMESTAMP()),
  updated_at = COALESCE(updated_at, CLOCK_TIMESTAMP()),
  revision = COALESCE(revision, 1);

WITH ranked AS (
  SELECT
    ctid,
    ROW_NUMBER() OVER (
      PARTITION BY data_key
      ORDER BY updated_at DESC NULLS LAST, created_at DESC NULLS LAST, revision DESC NULLS LAST
    ) AS row_num
  FROM public.app_data
)
DELETE FROM public.app_data app
USING ranked
WHERE app.ctid = ranked.ctid
  AND ranked.row_num > 1;

DO $$
BEGIN
  IF NOT EXISTS (
    SELECT 1
    FROM pg_constraint
    WHERE conrelid = 'public.app_data'::REGCLASS
      AND contype = 'p'
  ) THEN
    ALTER TABLE public.app_data
      ADD CONSTRAINT app_data_pkey PRIMARY KEY (data_key);
  END IF;
END;
$$;

ALTER TABLE public.app_data
  ALTER COLUMN data_key SET NOT NULL,
  ALTER COLUMN data_value SET DEFAULT '{}'::JSONB,
  ALTER COLUMN data_value SET NOT NULL,
  ALTER COLUMN created_at SET DEFAULT CLOCK_TIMESTAMP(),
  ALTER COLUMN created_at SET NOT NULL,
  ALTER COLUMN updated_at SET DEFAULT CLOCK_TIMESTAMP(),
  ALTER COLUMN updated_at SET NOT NULL,
  ALTER COLUMN revision SET DEFAULT 1,
  ALTER COLUMN revision SET NOT NULL;

CREATE INDEX IF NOT EXISTS app_data_updated_at_idx
  ON public.app_data (updated_at DESC);

-- Always bump the timestamp and revision when a device saves a planner bucket.
CREATE OR REPLACE FUNCTION public.set_app_data_sync_fields()
RETURNS TRIGGER
LANGUAGE plpgsql
SET search_path = public
AS $$
BEGIN
  NEW.data_key = BTRIM(NEW.data_key);
  NEW.data_value = COALESCE(NEW.data_value, '{}'::JSONB);

  IF TG_OP = 'INSERT' THEN
    NEW.created_at = COALESCE(NEW.created_at, CLOCK_TIMESTAMP());
    NEW.updated_at = COALESCE(NEW.updated_at, NEW.created_at, CLOCK_TIMESTAMP());
    NEW.revision = COALESCE(NEW.revision, 1);
  ELSE
    NEW.created_at = OLD.created_at;
    NEW.updated_at = CLOCK_TIMESTAMP();
    NEW.revision = COALESCE(OLD.revision, 0) + 1;
  END IF;

  RETURN NEW;
END;
$$;

DROP TRIGGER IF EXISTS set_app_data_updated_at ON public.app_data;
DROP TRIGGER IF EXISTS set_app_data_sync_fields ON public.app_data;

CREATE TRIGGER set_app_data_sync_fields
  BEFORE INSERT OR UPDATE ON public.app_data
  FOR EACH ROW
  EXECUTE FUNCTION public.set_app_data_sync_fields();

-- The current app uses the Supabase anon key directly, so anon needs read/write access.
-- This is intended for a personal single-user planner project.
ALTER TABLE public.app_data ENABLE ROW LEVEL SECURITY;

DROP POLICY IF EXISTS "Planner sync anon access" ON public.app_data;
DROP POLICY IF EXISTS "Planner sync read" ON public.app_data;
DROP POLICY IF EXISTS "Planner sync insert" ON public.app_data;
DROP POLICY IF EXISTS "Planner sync update" ON public.app_data;
DROP POLICY IF EXISTS "Planner sync delete" ON public.app_data;

CREATE POLICY "Planner sync read"
  ON public.app_data
  FOR SELECT
  TO anon, authenticated
  USING (true);

CREATE POLICY "Planner sync insert"
  ON public.app_data
  FOR INSERT
  TO anon, authenticated
  WITH CHECK (true);

CREATE POLICY "Planner sync update"
  ON public.app_data
  FOR UPDATE
  TO anon, authenticated
  USING (true)
  WITH CHECK (true);

CREATE POLICY "Planner sync delete"
  ON public.app_data
  FOR DELETE
  TO anon, authenticated
  USING (true);

GRANT USAGE ON SCHEMA public TO anon, authenticated;
GRANT SELECT, INSERT, UPDATE, DELETE ON public.app_data TO anon, authenticated;

SELECT 'Grounded planner sync table is ready' AS status;
